281inline Vec3D Vec3D::cross(Vec3D const& v) const
282{
283 Vec3D w;
284
285 w.r[0] = r[1] * v.r[2] - r[2] * v.r[1];
286 w.r[1] = r[2] * v.r[0] - r[0] * v.r[2];
287 w.r[2] = r[0] * v.r[1] - r[1] * v.r[0];
288
289 return w;
290}
