58  else if (cutoffType == CT_TANHU)

60  fPtr = &CutoffFunction:: fTANHU;

62  fdfPtr = &CutoffFunction::fdfTANHU;
63  }
64  else if (cutoffType == CT_TANH)
65  {
66  fPtr = &CutoffFunction:: fTANH;
67  dfPtr = &CutoffFunction:: dfTANH;
68  fdfPtr = &CutoffFunction::fdfTANH;

72  cutoffType == CT_POLY3 ||
73  cutoffType == CT_POLY4 ||
74  cutoffType == CT_EXP)
75  {

102 void CutoffFunction::setCutoffParameter(double const alpha)
103 {
104  if (alpha < 0.0 || alpha >= 1.0)
105  {
106  throw invalid_argument("ERROR: 0 <= alpha < 1.0 is required.\n");
107  }
108  this->alpha = alpha;
109  rci = rc * alpha;
110  iw = 1.0 / (rc - rci);
111  return;

114 double CutoffFunction::fCOS(double r) const
115 {
116  if (r < rci) return 1.0;
117  double const x = (r - rci) * iw;
118  return 0.5 * (cos(PI * x) + 1.0);
119 }

121 double CutoffFunction::dfCOS(double r) const
122 {
123  if (r < rci) return 0.0;
124  double const x = (r - rci) * iw;
125  return -PI_2 * iw * sin(PI * x);

128 void CutoffFunction::fdfCOS(double r, double& fc, double& dfc) const
129 {
130  if (r < rci)
131  {
132  fc = 1.0;
133  dfc = 0.0;
134  return;
135  }
136  double const x = (r - rci) * iw;
137  double const temp = cos(PI * x);
138  fc = 0.5 * (temp + 1.0);
139  dfc = -0.5 * iw * PI * sqrt(1.0 - temp * temp);
140  return;
141 }
