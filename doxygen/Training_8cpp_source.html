81 void Training::selectSets()
82 {
83  log << "\n";
84  log << "*** DEFINE TRAINING/TEST SETS ***********"
85  "**************************************\n";
86  log << "\n";
87 
88  vector<string> pCheck = {"force", "charge"};
89  bool check = false;
90  for (auto k : pk)
91  {
92  check |= (find(pCheck.begin(), pCheck.end(), k) != pCheck.end());
93  }
94  vector<size_t> numAtomsPerElement(numElements, 0);
95 
96  double testSetFraction = atof(settings["test_fraction"].c_str());
97  log << strpr("Desired test set ratio : %f\n", testSetFraction);
98  if (structures.size() > 0) hasStructures = true;
99  else hasStructures = false;
100 
101  string k;
102  for (size_t i = 0; i < structures.size(); ++i)
103  {
104  Structure& s = structures.at(i);
105  if (gsl_rng_uniform(rng) < testSetFraction)
106  {
107  s.sampleType = Structure::ST_TEST;
108  size_t const& na = s.numAtoms;
109  k = "energy"; if (p.exists(k)) p[k].numTestPatterns++;
110  k = "force"; if (p.exists(k)) p[k].numTestPatterns += 3 * na;
111  k = "charge"; if (p.exists(k)) p[k].numTestPatterns += na;
112  }
113  else
114  {
115  s.sampleType = Structure::ST_TRAINING;
116  for (size_t j = 0; j < numElements; ++j)
117  {
118  numAtomsPerElement.at(j) += s.numAtomsPerElement.at(j);
119  }
120 
121  k = "energy";
122  if (p.exists(k))
123  {
124  p[k].numTrainPatterns++;
125  p[k].updateCandidates.push_back(UpdateCandidate());
126  p[k].updateCandidates.back().s = i;
127  }
128  k = "force";
129  if (p.exists(k))
130  {
131  p[k].numTrainPatterns += 3 * s.numAtoms;
132  for (vector<Atom>::const_iterator it = s.atoms.begin();
133  it != s.atoms.end(); ++it)
134  {
135  for (size_t j = 0; j < 3; ++j)
136  {
137  p[k].updateCandidates.push_back(UpdateCandidate());
138  p[k].updateCandidates.back().s = i;
139  p[k].updateCandidates.back().a = it->index;
140  p[k].updateCandidates.back().c = j;
141  }
142  }
143  }
144  k = "charge";
145  if (p.exists(k))
146  {
147  p[k].numTrainPatterns += s.numAtoms;
148  for (vector<Atom>::const_iterator it = s.atoms.begin();
149  it != s.atoms.end(); ++it)
150  {
151  p[k].updateCandidates.push_back(UpdateCandidate());
152  p[k].updateCandidates.back().s = i;
153  p[k].updateCandidates.back().a = it->index;
154  }
155  }
156  }
157  }
158  for (size_t i = 0; i < numElements; ++i)
159  {
160  if (hasStructures && (numAtomsPerElement.at(i) == 0))
161  {
162  log << strpr("WARNING: Process %d has no atoms of element "
163  "%d (%2s).\n",
164  myRank,
165  i,
166  elementMap[i].c_str());
167  }
168  }
169  for (auto k : pk)
170  {
171  MPI_Allreduce(MPI_IN_PLACE, &(p[k].numTrainPatterns), 1, MPI_SIZE_T, MPI_SUM, comm);
172  MPI_Allreduce(MPI_IN_PLACE, &(p[k].numTestPatterns) , 1, MPI_SIZE_T, MPI_SUM, comm);
173  double sum = p[k].numTrainPatterns + p[k].numTestPatterns;
174  log << "Training/test split of data set for property \"" + k + "\":\n";
175  log << strpr("- Total patterns : %.0f\n", sum);
176  log << strpr("- Training patterns : %d\n", p[k].numTrainPatterns);
177  log << strpr("- Test patterns : %d\n", p[k].numTestPatterns);
178  log << strpr("- Test set fraction : %f\n", p[k].numTestPatterns / sum);
179  }
180 
181  log << "*****************************************"
182  "**************************************\n";
183 
184  return;
185 }

409  writeWeights("charge", "weightse.%03zu.norm");
410  }
411 
412  ofstream fileEvsV;
413  fileEvsV.open(strpr("evsv.dat.%04d", myRank).c_str());
414  if (myRank == 0)
415  {
416  // File header.

427  colInfo.push_back("Reference energy per atom.");

439  colInfo.push_back("Reference energy of structure (including offset).");
440  appendLinesToFile(fileEvsV,
441  createFileHeader(title, colSize, colName, colInfo));
442  }
443 
444  size_t numAtomsTotal = 0;
445  size_t numStructures = 0;
446  double meanEnergyPerAtomRef = 0.0;
447  double meanEnergyPerAtomNnp = 0.0;
448  double sigmaEnergyPerAtomRef = 0.0;
449  double sigmaEnergyPerAtomNnp = 0.0;
450  double meanForceRef = 0.0;
451  double meanForceNnp = 0.0;
452  double sigmaForceRef = 0.0;
453  double sigmaForceNnp = 0.0;
454  for (auto& s : structures)
455  {
456  // File output for evsv.dat.
457  fileEvsV << strpr("%16.8E %16.8E %10zu %16.8E %16.8E %16.8E\n",
458  s.volume / s.numAtoms,
459  s.energyRef / s.numAtoms,
460  s.numAtoms,
461  s.volume,
462  s.energyRef,
463  getEnergyWithOffset(s, true));
464  s.calculateNeighborList(maxCutoffRadius);
465 #ifdef N2P2_NO_SF_GROUPS
466  calculateSymmetryFunctions(s, true);
467 #else
468  calculateSymmetryFunctionGroups(s, true);
469 #endif
470  calculateAtomicNeuralNetworks(s, true);
471  calculateEnergy(s);
472  if (useForcesLocal) calculateForces(s);
473  s.clearNeighborList();
474 
475  numStructures++;
476  numAtomsTotal += s.numAtoms;
477  meanEnergyPerAtomRef += s.energyRef / s.numAtoms;
478  meanEnergyPerAtomNnp += s.energy / s.numAtoms;
479  for (auto& a : s.atoms)
480  {
481  meanForceRef += a.fRef[0] + a.fRef[1] + a.fRef[2];
482  meanForceNnp += a.f [0] + a.f [1] + a.f [2];
483  }
484  }
485 
486  fileEvsV.flush();
487  fileEvsV.close();
488  MPI_Barrier(MPI_COMM_WORLD);
489  log << "Writing energy/atom vs. volume/atom data to \"evsv.dat\".\n";
490  if (myRank == 0) combineFiles("evsv.dat");
491  MPI_Allreduce(MPI_IN_PLACE, &numStructures , 1, MPI_SIZE_T, MPI_SUM, MPI_COMM_WORLD);
492  MPI_Allreduce(MPI_IN_PLACE, &numAtomsTotal , 1, MPI_SIZE_T, MPI_SUM, MPI_COMM_WORLD);
493  MPI_Allreduce(MPI_IN_PLACE, &meanEnergyPerAtomRef, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
494  MPI_Allreduce(MPI_IN_PLACE, &meanEnergyPerAtomNnp, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
495  MPI_Allreduce(MPI_IN_PLACE, &meanForceRef , 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
496  MPI_Allreduce(MPI_IN_PLACE, &meanForceNnp , 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
497  meanEnergyPerAtomRef /= numStructures;
498  meanEnergyPerAtomNnp /= numStructures;
499  meanForceRef /= 3 * numAtomsTotal;
500  meanForceNnp /= 3 * numAtomsTotal;
501  for (auto const& s : structures)
502  {
503  double ediffRef = s.energyRef / s.numAtoms - meanEnergyPerAtomRef;
504  double ediffNnp = s.energy / s.numAtoms - meanEnergyPerAtomNnp;
505  sigmaEnergyPerAtomRef += ediffRef * ediffRef;
506  sigmaEnergyPerAtomNnp += ediffNnp * ediffNnp;
507  for (auto const& a : s.atoms)
508  {
509  double fdiffRef = a.fRef[0] - meanForceRef;
510  double fdiffNnp = a.f [0] - meanForceNnp;
511  sigmaForceRef += fdiffRef * fdiffRef;
512  sigmaForceNnp += fdiffNnp * fdiffNnp;
513  fdiffRef = a.fRef[1] - meanForceRef;
514  fdiffNnp = a.f [1] - meanForceNnp;
515  sigmaForceRef += fdiffRef * fdiffRef;
516  sigmaForceNnp += fdiffNnp * fdiffNnp;
517  fdiffRef = a.fRef[2] - meanForceRef;
518  fdiffNnp = a.f [2] - meanForceNnp;
519  sigmaForceRef += fdiffRef * fdiffRef;
520  sigmaForceNnp += fdiffNnp * fdiffNnp;
521  }
522  }
523  MPI_Allreduce(MPI_IN_PLACE, &sigmaEnergyPerAtomRef, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
524  MPI_Allreduce(MPI_IN_PLACE, &sigmaEnergyPerAtomNnp, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
525  MPI_Allreduce(MPI_IN_PLACE, &sigmaForceRef , 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
526  MPI_Allreduce(MPI_IN_PLACE, &sigmaForceNnp , 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
527  sigmaEnergyPerAtomRef = sqrt(sigmaEnergyPerAtomRef / (numStructures - 1));
528  sigmaEnergyPerAtomNnp = sqrt(sigmaEnergyPerAtomNnp / (numStructures - 1));
529  sigmaForceRef = sqrt(sigmaForceRef / (3 * numAtomsTotal - 1));
530  sigmaForceNnp = sqrt(sigmaForceNnp / (3 * numAtomsTotal - 1));
531  log << "\n";
532  log << strpr("Total number of structures : %zu\n", numStructures);
533  log << strpr("Total number of atoms : %zu\n", numAtomsTotal);
534  log << "----------------------------------\n";
535  log << "Reference data statistics:\n";
536  log << "----------------------------------\n";
537  log << strpr("Mean/sigma energy per atom : %16.8E +/- %16.8E\n",
538  meanEnergyPerAtomRef,
539  sigmaEnergyPerAtomRef);
540  log << strpr("Mean/sigma force : %16.8E +/- %16.8E\n",
541  meanForceRef,
542  sigmaForceRef);
543  log << "----------------------------------\n";
544  log << "Initial NNP prediction statistics:\n";
545  log << "----------------------------------\n";
546  log << strpr("Mean/sigma energy per atom : %16.8E +/- %16.8E\n",
547  meanEnergyPerAtomNnp,
548  sigmaEnergyPerAtomNnp);
549  log << strpr("Mean/sigma force : %16.8E +/- %16.8E\n",
550  meanForceNnp,
551  sigmaForceNnp);
552  log << "----------------------------------\n";
553  // Now set conversion quantities of Mode class.
554  if (settings["normalize_data_set"] == "stats-only")
555  {
556  log << "Data set statistics computation completed, now make up for \n";
557  log << "initially skipped normalization setup...\n";
558  log << "\n";
559  setupNormalization(false);
560  }
561  else if (settings["normalize_data_set"] == "ref")
562  {
563  log << "Normalization based on standard deviation of reference data "
564  "selected:\n";
565  log << "\n";
566  log << " mean(e_ref) = 0, sigma(e_ref) = sigma(F_ref) = 1\n";
567  log << "\n";
568  meanEnergy = meanEnergyPerAtomRef;
569  convEnergy = 1.0 / sigmaEnergyPerAtomRef;
570  if (useForcesLocal) convLength = sigmaForceRef / sigmaEnergyPerAtomRef;
571  else convLength = 1.0;
572  normalize = true;
573  }
574  else if (settings["normalize_data_set"] == "force")
575  {
576  if (!useForcesLocal)
577  {
578  throw runtime_error("ERROR: Selected normalization mode only "
579  "possible when forces are available.\n");
580  }
581  log << "Normalization based on standard deviation of reference forces "
582  "and their\n";
583  log << "initial prediction selected:\n";
584  log << "\n";
585  log << " mean(e_ref) = 0, sigma(F_NNP) = sigma(F_ref) = 1\n";
586  log << "\n";
587  meanEnergy = meanEnergyPerAtomRef;
588  convEnergy = sigmaForceNnp / sigmaForceRef;
589  convLength = sigmaForceNnp;
590  normalize = true;
591  }
592  else
593  {
594  throw runtime_error("ERROR: Unknown data set normalization mode.\n");
595  }
596 
597  if (settings["normalize_data_set"] != "stats-only")
598  {
599  log << "Final conversion data:\n";
600  log << strpr("Mean ref. energy per atom = %24.16E\n", meanEnergy);
601  log << strpr("Conversion factor energy = %24.16E\n", convEnergy);
602  log << strpr("Conversion factor length = %24.16E\n", convLength);
603  log << "----------------------------------\n";
604  }
605 
606  if ((myRank == 0) &&
607  (settings["normalize_data_set"] != "stats-only"))
608  {
609  log << "\n";
610  log << "Writing backup of original settings file to "
611  "\"input.nn.bak\".\n";
612  ofstream fileSettings;
613  fileSettings.open("input.nn.bak");
614  writeSettingsFile(&fileSettings);
615  fileSettings.close();
616 
617  log << "\n";
618  log << "Writing normalization data to settings file \"input.nn\".\n";
