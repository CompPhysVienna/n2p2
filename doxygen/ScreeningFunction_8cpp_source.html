31void ScreeningFunction::setInnerOuter(double inner, double outer)
32{
33 if (inner >= outer)
34 {
35 throw invalid_argument("ERROR: Inner radius of transition region >= "
36 "outer radius.\n");
37 }
38
39 this->inner = inner;
40 this->outer = outer;
41 this->scale = 1.0 / (outer - inner);
42
43 return;
44}

46void ScreeningFunction::changeLengthUnits(double const conv)
47{
48 inner *= conv;
49 outer *= conv;
50 scale /= conv;
