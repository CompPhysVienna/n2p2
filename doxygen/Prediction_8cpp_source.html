44void Prediction::readStructureFromFile(string const& fileName)
45{
46 ifstream file;
47 file.open(fileName.c_str());
48 structure.reset();
49 structure.setElementMap(elementMap);
50 structure.readFromFile(fileName);
51 removeEnergyOffset(structure);
52 if (normalize)
53 {
54 structure.toNormalizedUnits(meanEnergy, convEnergy, convLength);
55 }
56 file.close();
57
58 return;
59}

61void Prediction::predict()
62{
63 structure.calculateNeighborList(maxCutoffRadius);
64#ifdef N2P2_NO_SF_GROUPS
65 calculateSymmetryFunctions(structure, true);
66#else
67 calculateSymmetryFunctionGroups(structure, true);
68#endif
69 calculateAtomicNeuralNetworks(structure, true);
70 calculateEnergy(structure);
71 if (nnpType == NNPType::SHORT_CHARGE_NN) calculateCharge(structure);
72 calculateForces(structure);
73 if (normalize)
74 {
75 structure.toPhysicalUnits(meanEnergy, convEnergy, convLength);
76 }
77 addEnergyOffset(structure, false);
78 addEnergyOffset(structure, true);
79
80 return;
81}
